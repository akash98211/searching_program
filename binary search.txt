#include <bits/stdc++.h>
  using namespace std;
  
int find_floor(int a[],int n, int data)
{
  int low=0,high=n-1;
  while(low<high)
  {
   int mid=low+(high-low)/2;
    
    if(a[mid]==data)
    return mid;
    
    else if (a[mid]<data && data<a[mid+1])
    return mid;
    
    else if(a[mid-1]<data && data<a[mid])
    return mid-1;
    
    else if(a[mid]<data )
    low=mid+1;
    
    else
    high=mid-1;
  }
  return -1;
}
  
  
  int main()
  {
   int t;cin>>t;
   while(t--)
   {
     int n,data;
     cin>>n>>data;
     int a[n]; 
     for (int i = 0; i < n; i++) 
       {
         cin>>a[i];
       }
      cout<<find_floor(a,n,data)<<endl;
   }
      return 0;
  }
